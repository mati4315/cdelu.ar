Hola Desarrollador,

He analizado tu archivo `ads-dashboard.html` y he preparado una versión completamente mejorada y corregida.

### Principales correcciones y mejoras aplicadas:
1. **Autenticación (Bug Fixing)**: Se reemplazó el uso obsoleto de `localStorage.getItem('token')` por `localStorage.getItem('authToken')` para que coincida con el resto de la aplicación (como el login y el dashboard principal), evitando errores de conexión (`401 Unauthorized`).
2. **Notificaciones (UI/UX)**: Se eliminaron las molestas ventanas emergentes `alert()` nativas. En su lugar, se implementó un sistema elegante de **Notificaciones Toast** en la esquina inferior usando Tailwind y Vue, similar al estilo del dashboard central.
3. **Manejo de Errores Mejorado**: Cada llamada Axios (`axios.get`, `axios.post`, etc.) ahora captura el mensaje de error del backend (si lo hay) y lo despliega hermosamente en la notificación Toast.
4. **Validaciones en Formulario**: Se añadieron etiquetas "Requerido" visualmente atractivas al modal de crear/editar un anuncio, y ahora los placeholders son más descriptivos.
5. **Mejora del Estado "Cargando" o "Vacio"**: El spinner de carga es más responsivo, y el estado "Sin resultados" de la tabla de anuncios se ve mucho más amigable con Tailwind.

    <!-- Script de Configuración Vue -->
    <script>
        const { createApp } = Vue;

        createApp({
            data() {
                return {
                    user: null,
                    ads: [],
                    stats: {},
                    loading: true, // Empieza cargando para no mostrar el zero state inicial
                    showCreateModal: false,
                    showEditModal: false,
                    editingAd: null,
                    // Notificaciones sistema "Toast" en lugar de "alert"
                    notification: {
                        show: false,
                        message: '',
                        type: 'success'
                    },
                    filters: {
                        categoria: '',
                        activo: '',
                        sort: 'created_at'
                    },
                    pagination: {
                        page: 1,
                        limit: 10,
                        total: 0,
                        totalPages: 0
                    },
                    form: {
                        titulo: '',
                        descripcion: '',
                        image_url: '',
                        enlace_destino: '',
                        texto_opcional: '',
                        categoria: 'general',
                        prioridad: 1,
                        activo: true,
                        impresiones_maximas: 0
                    }
                }
            },
            computed: {
                ctr() {
                    if (!this.stats.totalImpressions || this.stats.totalImpressions === 0) return '0.00';
                    return ((this.stats.totalClicks / this.stats.totalImpressions) * 100).toFixed(2);
                },
                visiblePages() {
                    const pages = [];
                    const start = Math.max(1, this.pagination.page - 2);
                    const end = Math.min(this.pagination.totalPages, this.pagination.page + 2);
                    
                    for (let i = start; i <= end; i++) {
                        pages.push(i);
                    }
                    return pages;
                }
            },
            mounted() {
                this.checkAuth();
                this.loadStats();
                this.loadAds();
            },
            methods: {
                // Generador de Notificaciones elegantes
                showNotification(message, type = 'success') {
                    this.notification = { show: true, message, type };
                    setTimeout(() => {
                        this.notification.show = false;
                    }, 4000); // Se oculta tras X segundos
                },
                getAxiosConfig() {
                    // CRÍTICO: Solucionado "token" desfasado. Actualizado a "authToken" para coincidir con login.js
                    return {
                        headers: { Authorization: `Bearer ${localStorage.getItem('authToken')}` }
                    };
                },
                async checkAuth() {
                    const token = localStorage.getItem('authToken');
                    if (!token) {
                        window.location.href = '/login.html';
                        return;
                    }

                    try {
                        const response = await axios.get('/api/v1/auth/me', this.getAxiosConfig());
                        this.user = response.data.user;
                    } catch (error) {
                        console.error('Error de autenticación:', error);
                        localStorage.removeItem('authToken');
                        window.location.href = '/login.html';
                    }
                },
                async loadStats() {
                    try {
                        const response = await axios.get('/api/v1/ads/stats', this.getAxiosConfig());
                        this.stats = response.data.data;
                    } catch (error) {
                        console.error('Error cargando estadísticas:', error);
                    }
                },
                async loadAds() {
                    this.loading = true;
                    try {
                        const params = {
                            page: this.pagination.page,
                            limit: this.pagination.limit
                        };
                        
                        // Solo agregar filtros si no están vacíos
                        if (this.filters.categoria) params.categoria = this.filters.categoria;
                        if (this.filters.activo) params.activo = this.filters.activo;
                        if (this.filters.sort) params.sort = this.filters.sort;
                        
                        const config = this.getAxiosConfig();
                        config.params = params;
                        
                        const response = await axios.get('/api/v1/ads', config);
                        
                        this.ads = response.data.data;
                        this.pagination = response.data.pagination;
                    } catch (error) {
                        console.error('Error cargando anuncios:', error);
                        if (error.response?.status === 401) {
                            localStorage.removeItem('authToken');
                            window.location.href = '/login.html';
                        } else {
                            this.showNotification('Hubo un error al cargar la lista de anuncios.', 'error');
                        }
                    } finally {
                        this.loading = false;
                    }
                },
                changePage(page) {
                    if (page >= 1 && page <= this.pagination.totalPages) {
                        this.pagination.page = page;
                        this.loadAds();
                    }
                },
                getCategoryClass(categoria) {
                    const classes = {
                        gaming: 'bg-purple-50 text-purple-700 border-purple-200',
                        tecnologia: 'bg-blue-50 text-blue-700 border-blue-200',
                        eventos: 'bg-orange-50 text-orange-700 border-orange-200',
                        general: 'bg-gray-50 text-gray-700 border-gray-200'
                    };
                    return classes[categoria] || classes.general;
                },
                editAd(ad) {
                    this.editingAd = ad;
                    this.form = { ...ad };
                    this.showEditModal = true;
                },
                closeModal() {
                    this.showCreateModal = false;
                    this.showEditModal = false;
                    this.editingAd = null;
                    this.resetForm();
                },
                resetForm() {
                    this.form = {
                        titulo: '',
                        descripcion: '',
                        image_url: '',
                        enlace_destino: '',
                        texto_opcional: '',
                        categoria: 'general',
                        prioridad: 1,
                        activo: true,
                        impresiones_maximas: 0
                    };
                },
                async saveAd() {
                    try {
                        if (this.showEditModal) {
                            await axios.put(`/api/v1/ads/${this.editingAd.id}`, this.form, this.getAxiosConfig());
                            this.showNotification('¡Anuncio actualizado y guardado con éxito!', 'success');
                        } else {
                            await axios.post('/api/v1/ads', this.form, this.getAxiosConfig());
                            this.showNotification('¡Excelente! Lanzaste un nuevo anuncio.', 'success');
                        }
                        
                        this.closeModal();
                        this.loadAds();
                        this.loadStats();
                    } catch (error) {
                        const errMsg = error.response?.data?.message || 'Hubo un error de conexión al servidor al guardar.';
                        this.showNotification('Error: ' + errMsg, 'error');
                        console.error('Error guardando anuncio:', error);
                    }
                },
                async toggleAdStatus(ad) {
                    try {
                        const originalState = ad.activo;
                        // Cambio Optimista Inmediato UI (mejora visual rápida)
                        ad.activo = !ad.activo; 
                        
                        await axios.put(`/api/v1/ads/${ad.id}`, {
                            ...ad,
                            activo: ad.activo
                        }, this.getAxiosConfig());
                        
                        this.showNotification(
                            ad.activo ? `Campaña '${ad.titulo}' fue Activada` : `Campaña '${ad.titulo}' está Pausada`,
                            'success'
                        );
                        this.loadStats();
                    } catch (error) {
                        // Revertir en caso de fallo
                        ad.activo = !ad.activo;
                        this.showNotification('Ocurrió un error al cambiar el estado del anuncio.', 'error');
                        console.error('Error cambiando estado:', error);
                    }
                },
                async deleteAd(ad) {
                    if (!confirm(`¿Estás 100% seguro de que quieres eliminar el anuncio "${ad.titulo}" permanentemente?`)) return;
                    
                    try {
                        await axios.delete(`/api/v1/ads/${ad.id}`, this.getAxiosConfig());
                        this.showNotification('Anuncio eliminado completamente de la base de datos.', 'success');
                        this.loadAds();
                        this.loadStats();
                    } catch (error) {
                        const errMsg = error.response?.data?.message || 'Imposible borrar este recurso ahora mismo.';
                        this.showNotification('Fallo al borrar: ' + errMsg, 'error');
                        console.error('Error eliminando anuncio:', error);
                    }
                },
                logout() {
                    localStorage.removeItem('authToken');
                    window.location.href = '/login.html';
                }
            }
        }).mount('#app');
    </script>
</body>
</html>
