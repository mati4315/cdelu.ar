2. JS - Lógica de Funcionamiento (Añadir a la sección de scripts)
=======================================================

// === API DOCS LOGIC ===
const apiTokenInput = document.getElementById('apiToken');
const refreshApiTokenBtn = document.getElementById('refreshApiToken');
const apiMethodSelect = document.getElementById('apiMethod');
const apiEndpointSelect = document.getElementById('apiEndpointSelect');
const apiCustomEndpoint = document.getElementById('apiCustomEndpoint');
const apiBodyContainer = document.getElementById('apiBodyContainer');
const sendApiRequestBtn = document.getElementById('sendApiRequest');
const apiResponseOutput = document.getElementById('apiResponseOutput');
const apiResponseStatus = document.getElementById('apiResponseStatus');

// Refrescar token desde localStorage (authToken es el nombre usado en el dashboard)
refreshApiTokenBtn?.addEventListener('click', () => {
    const token = localStorage.getItem('authToken');
    if (token && apiTokenInput) {
        apiTokenInput.value = token;
        showNotification('Token actualizado desde la sesión actual', 'success');
    } else {
        showNotification('No se encontró un token de sesión activo', 'warning');
    }
});

// Cambiar visibilidad de campos según selección de método
apiMethodSelect?.addEventListener('change', function() {
    if (this.value === 'GET' || this.value === 'DELETE') {
        apiBodyContainer?.classList.add('d-none');
    } else {
        apiBodyContainer?.classList.remove('d-none');
    }
});

// Cambiar visibilidad de endpoint personalizado
apiEndpointSelect?.addEventListener('change', function() {
    if (this.value === 'custom') {
        apiCustomEndpoint?.classList.remove('d-none');
    } else {
        apiCustomEndpoint?.classList.add('d-none');
    }
});

// Ejecutar Petición API
sendApiRequestBtn?.addEventListener('click', async () => {
    const method = apiMethodSelect.value;
    const finalEndpoint = apiEndpointSelect.value === 'custom' ? apiCustomEndpoint.value : apiEndpointSelect.value;
    const token = apiTokenInput.value;
    const bodyText = document.getElementById('apiRequestBody').value;

    if (!finalEndpoint) {
        showNotification('Por favor ingresa un endpoint válido', 'warning');
        return;
    }

    apiResponseOutput.textContent = 'Enviando petición a ' + finalEndpoint + '...';
    apiResponseStatus.textContent = 'Cargando...';
    apiResponseStatus.className = 'badge bg-info';
    
    try {
        const options = {
            method,
            headers: { 
                'Authorization': `Bearer ${token}`,
                'Accept': 'application/json'
            }
        };

        if (method !== 'GET' && method !== 'DELETE') {
            options.headers['Content-Type'] = 'application/json';
            try {
                // Validar JSON antes de enviar
                JSON.parse(bodyText);
                options.body = bodyText;
            } catch (jsonErr) {
                throw new Error('El cuerpo de la petición no es un JSON válido');
            }
        }

        const response = await fetch(finalEndpoint, options);
        apiResponseStatus.textContent = `${response.status} ${response.statusText}`;
        apiResponseStatus.className = `badge ${response.ok ? 'bg-success' : 'bg-danger'}`;

        const data = await response.json().catch(() => null);
        apiResponseOutput.textContent = data ? JSON.stringify(data, null, 2) : 'La respuesta no es un JSON válido o está vacía.';
        
        if (response.ok) {
            showNotification('Petición completada con éxito', 'success');
        } else {
            showNotification('La API devolvió un error: ' + response.status, 'error');
        }
    } catch (error) {
        apiResponseOutput.textContent = `Error: ${error.message}`;
        apiResponseStatus.textContent = 'Error de Red';
        apiResponseStatus.className = 'badge bg-danger';
        showNotification(error.message, 'error');
    }
});
